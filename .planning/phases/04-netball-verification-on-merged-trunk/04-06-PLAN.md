<!-- Fairness tiers (from src/lib/sports/netball/fairness.ts) -->

5 tiers, applied per (player, candidate-position) pair:
1. **+100,000** — unplayed-third bonus (player has never played in this third this game/season)
2. **−50,000** — same-position penalty (player came off this position last quarter)
3. **−10,000** — same-third penalty (player was in the same third last quarter)
4. **−5,000 per overlap** — friend-pair split penalty (avoid keeping the same friend pair on court)
5. **season under-utilisation tiebreak** — bonus for players with low season-availability ratio

Tie-breaks:
1. lower `thisGameTotalMs` wins
2. lower seasonAvailability ratio wins
3. seeded shuffle (deterministic — uses team_id + game_id + quarter as seed)

// NETBALL-02 — Netball quarter-break rotation suggester end-to-end.
//
// The pure-function 5-tier logic is covered by
// src/lib/__tests__/netballFairness.test.ts. This spec proves the
// END-TO-END behaviour:
//   1. Q-break entry path works (Q1 auto-end → modal → Q-break shell)
//   2. The suggester runs and populates a 7-position lineup
//   3. The unplayed-third +100k tier is observable in the suggester's
//      output (a player who didn't play Q1 must be on court at Q2)
//   4. Pressing Start Q2 writes period_break_swap + quarter_start events
//
// Per CONTEXT D-CONTEXT-seed-strategy: prefer the Kotara Koalas seed
// when present (real season history), fall back to a fresh factory
// team with synthetic prior-game data when absent. The audit at
// e2e/helpers/seed-audit.ts decides which path to take.

import { test, expect } from "@playwright/test";
import { createAdminClient } from "../fixtures/supabase";
import { makeTeam, makePlayers, makeGame } from "../fixtures/factories";
import { auditKotaraKoalas, KOTARA_KOALAS_TEAM_ID } from "../helpers/seed-audit";

test.describe.configure({ mode: "parallel" });

const NETBALL_LINEUP_KEYS = ["gs", "ga", "wa", "c", "wd", "gd", "gk"] as const;

async function suppressWalkthrough(page: import("@playwright/test").Page) {
  await page.addInitScript(() => {
    try { localStorage.setItem("nb-walkthrough-seen", "1"); } catch {}
  });
}

/**
 * Sets up a fresh netball team + game with Q1 PAST the auto-hooter, so
 * mounting the live page auto-opens the QuarterEndModal. The lineup
 * uses 7 of the 9 players — players[7] and players[8] are LEFT OFF
 * COURT in Q1 to make the unplayed-third tier observable in Q2.
 */
async function setupQ1AutoEnded(opts: {
  trackScoring?: boolean;
  /** team.quarter_length_seconds override (smaller = faster auto-hooter). */
  teamQuarterLengthSeconds?: number;
}) {
  const admin = createAdminClient();
  const { data: superAdmin } = await admin.auth.admin.listUsers();
  const ownerId = superAdmin.users.find(
    (u) => u.email === process.env.TEST_SUPER_ADMIN_EMAIL,
  )!.id;

  const team = await makeTeam(admin, {
    ownerId,
    ageGroup: "go",
    sport: "netball",
    name: `NB-QB-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
  });
  const teamQS = opts.teamQuarterLengthSeconds ?? 480;   // default 8 min
  await admin
    .from("teams")
    .update({
      track_scoring: opts.trackScoring ?? false,
      quarter_length_seconds: teamQS,
    })
    .eq("id", team.id);

  const players = await makePlayers(admin, {
    teamId: team.id,
    ownerId,
    count: 9,
    ageGroup: "U10",
  });
  const game = await makeGame(admin, { teamId: team.id, ownerId, ageGroup: "U10" });

  // Q1 lineup: players[0..6] on court, players[7..8] benched (will get
  // the unplayed-third bonus when the suggester runs for Q2).
  const lineup: Record<string, string[]> = Object.fromEntries(
    NETBALL_LINEUP_KEYS.map((k, i) => [k, [players[i].id]]),
  );
  // Backdate quarter_start by (teamQS + 60)s so elapsed > quarterMs at
  // mount → auto-hooter fires → modal opens.
  const startedAt = new Date(Date.now() - (teamQS + 60) * 1000).toISOString();
  await admin.from("game_events").insert([
    { game_id: game.id, type: "lineup_set",   metadata: { lineup }, created_by: ownerId, created_at: startedAt },
    { game_id: game.id, type: "quarter_start", metadata: { quarter: 1 }, created_by: ownerId, created_at: startedAt },
  ]);
  await admin.from("games").update({ status: "in_progress" }).eq("id", game.id);

  return { team, players, game, ownerId, admin };
}

/**
 * Awaits the Q-break view. Netball auto-transitions through hooter
 * directly into the Q-break shell — there is no intermediate "Select
 * team for Q2" modal (that is AFL-only, from QuarterEndModal.tsx).
 * NetballLiveGame.tsx:184-198 fires endNetballQuarter when remainingMs
 * <= 0; the quarterEnded && currentQuarter < 4 branch at :959-1027
 * renders NetballQuarterBreak with no user click required. The
 * Q-break shell's "Start Q2" button is the canonical signal.
 */
async function enterQBreakView(page: import("@playwright/test").Page) {
  await expect(page.getByRole("button", { name: /start q2/i })).toBeVisible({ timeout: 10_000 });
}

test("NETBALL-02: Q-break shell renders with a 7-position suggested lineup after Q1 auto-ends", async ({ page }) => {
  const { team, game } = await setupQ1AutoEnded({});
  await suppressWalkthrough(page);
  await page.goto(`/teams/${team.id}/games/${game.id}/live`);
  await enterQBreakView(page);

  // The suggester ran on mount of the Q-break component. The "Apply
  // suggested reshuffle" button is the canonical evidence.
  await expect(
    page.getByRole("button", { name: /apply suggested reshuffle/i }),
  ).toBeVisible({ timeout: 5_000 });

  // The Court display in Q-break renders the 7 position labels (GS / GA
  // / WA / C / WD / GD / GK). At least one should be discoverable as
  // text content. (Position labels are rendered by NetballQuarterBreak's
  // Court via positionToken/CourtDisplay primitives.)
  for (const label of ["GS", "GA", "WA", "C", "WD", "GD", "GK"]) {
    await expect(page.getByText(new RegExp(`\\b${label}\\b`))).toBeVisible({ timeout: 3_000 });
  }
});

test("NETBALL-02: unplayed-third tier dominates — players who didn't play Q1 appear on court for Q2", async ({ page }) => {
  // Setup leaves players[7] + players[8] BENCHED in Q1. Per the +100k
  // unplayed-third tier, both should be placed on court at Q2 unless
  // overridden by tighter constraints (and with only 9 players for 7
  // slots, both unplayed players will be picked + one Q1 starter
  // benched).
  const { team, game, players } = await setupQ1AutoEnded({});
  await suppressWalkthrough(page);
  await page.goto(`/teams/${team.id}/games/${game.id}/live`);
  await enterQBreakView(page);

  // Click "Apply suggested reshuffle" to commit the suggester's lineup
  // to the visible Court (it's already on screen but applying makes the
  // Court reflect the suggester's exact layout — the toggle from
  // "Apply suggested reshuffle" → "✓ Using suggested reshuffle" is
  // visible too).
  await page.getByRole("button", { name: /apply suggested reshuffle/i }).click();

  // Both unplayed players should now appear in the rendered Court
  // (somewhere — exact position varies by tier scoring). Their names
  // are single-word per factory pool; lookup by name.
  await expect(page.getByText(new RegExp(players[7].full_name, "i"))).toBeVisible({ timeout: 3_000 });
  await expect(page.getByText(new RegExp(players[8].full_name, "i"))).toBeVisible();
});

test("NETBALL-02: Start Q2 writes period_break_swap + quarter_start events", async ({ page }) => {
  const { team, game, admin } = await setupQ1AutoEnded({});
  await suppressWalkthrough(page);
  await page.goto(`/teams/${team.id}/games/${game.id}/live`);
  await enterQBreakView(page);

  // Apply the suggested reshuffle so Start Q2 has a committed lineup.
  await page.getByRole("button", { name: /apply suggested reshuffle/i }).click();
  await page.getByRole("button", { name: /start q2/i }).click();

  // Wait for the live state to mount (header shows Q2 label).
  await expect(page.getByText(/^q2\b/i)).toBeVisible({ timeout: 10_000 });

  // DB events: period_break_swap (for Q1→Q2) + quarter_start (Q2)
  await expect.poll(
    async () => {
      const { data } = await admin
        .from("game_events")
        .select("type, metadata")
        .eq("game_id", game.id);
      const events = (data ?? []) as { type: string; metadata: { quarter?: number } }[];
      const hasSwap = events.some((e) => e.type === "period_break_swap");
      const hasQ2Start = events.some((e) => e.type === "quarter_start" && e.metadata?.quarter === 2);
      return hasSwap && hasQ2Start;
    },
    { timeout: 5_000, intervals: [200, 200, 500, 500, 1000] },
  ).toBe(true);
});

// ─── Kotara Koalas / season-history audit (TEST-05 hook) ───
//
// Optional fourth case: when Kotara Koalas is present, exercise a Q-break
// against a Kotara game so the season-utilisation tie-break has REAL
// 5-game history to read against. When absent, log a skip so Phase 5
// can capture the gap. This test is skip-when-absent rather than
// fail-when-absent because TEST-05 is a Phase 5 acceptance, not Phase 4.

test("NETBALL-02 (Kotara optional): suggester runs against Kotara season history when seed is present", async ({ page }) => {
  const admin = createAdminClient();
  const audit = await auditKotaraKoalas(admin);
  if (!audit.present) {
    test.skip(true, `Kotara Koalas seed absent (gameCount=${audit.gameCount}, playerCount=${audit.playerCount}). TEST-05 will surface this in Phase 5; not blocking Phase 4.`);
    return;
  }

  // Create a NEW game for Kotara (don't mutate existing finalised games).
  const { data: superAdmin } = await admin.auth.admin.listUsers();
  const ownerId = superAdmin.users.find(
    (u) => u.email === process.env.TEST_SUPER_ADMIN_EMAIL,
  )!.id;

  // Confirm the team owner is queryable, and pull Kotara's actual roster.
  const { data: kotaraPlayers, error: playerErr } = await admin
    .from("players")
    .select("id, full_name")
    .eq("team_id", KOTARA_KOALAS_TEAM_ID)
    .eq("is_active", true)
    .order("jersey_number");
  if (playerErr || !kotaraPlayers || kotaraPlayers.length < 7) {
    test.skip(true, `Kotara Koalas player roster unusable: ${playerErr?.message ?? `count=${kotaraPlayers?.length ?? 0}`}`);
    return;
  }

  // Create a fresh game for Kotara.
  const game = await makeGame(admin, {
    teamId: KOTARA_KOALAS_TEAM_ID,
    ownerId,
    ageGroup: "U10",
    opponent: `NB-QB-Kotara-${Date.now()}`,
  });
  const lineup: Record<string, string[]> = Object.fromEntries(
    NETBALL_LINEUP_KEYS.map((k, i) => [k, [kotaraPlayers[i].id]]),
  );
  const startedAt = new Date(Date.now() - 540_000).toISOString();   // 9 min ago — past 8-min default
  await admin.from("game_events").insert([
    { game_id: game.id, type: "lineup_set",   metadata: { lineup }, created_by: ownerId, created_at: startedAt },
    { game_id: game.id, type: "quarter_start", metadata: { quarter: 1 }, created_by: ownerId, created_at: startedAt },
  ]);
  await admin.from("games").update({ status: "in_progress" }).eq("id", game.id);

  await suppressWalkthrough(page);
  await page.goto(`/teams/${KOTARA_KOALAS_TEAM_ID}/games/${game.id}/live`);
  await enterQBreakView(page);

  // The suggester ran with Kotara's full season history feeding the
  // tier-5 utilisation tiebreak. Just verify the Q-break shell rendered.
  await expect(
    page.getByRole("button", { name: /apply suggested reshuffle/i }),
  ).toBeVisible({ timeout: 5_000 });
});
