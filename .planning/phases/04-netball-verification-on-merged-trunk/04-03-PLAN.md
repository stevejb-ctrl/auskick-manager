// NETBALL-05 — Netball stats dashboard renders 5 sections with per-
// position breakdown; AFL aggregator output does NOT leak when sport
// is netball.
//
// Setup pattern: seed a 4-quarter finalised netball game directly via
// admin client (writing lineup_set, quarter_start/end events,
// period_break_swap, score events with player_id, then flipping
// games.status to 'completed'). Visit the stats route. Assert section
// headings.
//
// Covers: src/app/(app)/teams/[teamId]/stats/page.tsx sport-branch
//         dispatch (line 109); src/components/dashboard/
//         NetballDashboardShell.tsx 5-section render.

import { test, expect } from "@playwright/test";
import { createAdminClient } from "../fixtures/supabase";
import { makeTeam, makePlayers, makeGame } from "../fixtures/factories";

test.describe.configure({ mode: "parallel" });

const NETBALL_LINEUP_KEYS = ["gs", "ga", "wa", "c", "wd", "gd", "gk"] as const;

async function seedFinalisedNetballGame(opts: { trackScoring: boolean }) {
  const admin = createAdminClient();
  const { data: superAdmin } = await admin.auth.admin.listUsers();
  const ownerId = superAdmin.users.find(
    (u) => u.email === process.env.TEST_SUPER_ADMIN_EMAIL,
  )!.id;

  const team = await makeTeam(admin, {
    ownerId,
    ageGroup: "go",
    sport: "netball",
    name: `NB-Stats-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
  });
  await admin
    .from("teams")
    .update({ track_scoring: opts.trackScoring })
    .eq("id", team.id);

  const players = await makePlayers(admin, {
    teamId: team.id,
    ownerId,
    count: 9,           // netball squad
    ageGroup: "U10",    // dodge AFL-only AGE_GROUPS lookup; only used for default count, which we override
  });
  const game = await makeGame(admin, { teamId: team.id, ownerId, ageGroup: "U10" });

  // Build a 7-position netball lineup; bench remaining players
  const lineup: Record<string, string[]> = Object.fromEntries(
    NETBALL_LINEUP_KEYS.map((k, i) => [k, [players[i].id]]),
  );
  // Backdate so Q1 < Q2 < Q3 < Q4 < now in created_at order.
  const base = Date.now() - 60 * 60 * 1000; // 1 hour ago
  const ts = (offsetMin: number) => new Date(base + offsetMin * 60_000).toISOString();

  const events = [
    { type: "lineup_set",   metadata: { lineup }, created_at: ts(0) },
    { type: "quarter_start", metadata: { quarter: 1 }, created_at: ts(1) },
    // Goals only when track_scoring=true. If false, leave the goals lines
    // empty so the chemistry section still has time-on-court data but no
    // scoring data — this also exercises the "hasScoringData=false" branch.
    ...(opts.trackScoring
      ? [
          { type: "score", metadata: { kind: "goal", side: "team", quarter: 1 }, player_id: players[0].id, created_at: ts(5) },
          { type: "score", metadata: { kind: "goal", side: "opp",  quarter: 1 }, created_at: ts(6) },
        ]
      : []),
    { type: "quarter_end",   metadata: { quarter: 1 }, created_at: ts(10) },
    { type: "quarter_start", metadata: { quarter: 2 }, created_at: ts(11) },
    { type: "quarter_end",   metadata: { quarter: 2 }, created_at: ts(20) },
    { type: "quarter_start", metadata: { quarter: 3 }, created_at: ts(21) },
    { type: "quarter_end",   metadata: { quarter: 3 }, created_at: ts(30) },
    { type: "quarter_start", metadata: { quarter: 4 }, created_at: ts(31) },
    { type: "quarter_end",   metadata: { quarter: 4 }, created_at: ts(40) },
  ].map((e) => ({
    ...e,
    game_id: game.id,
    created_by: ownerId,
  }));

  const { error } = await admin.from("game_events").insert(events);
  if (error) throw new Error(`seedFinalisedNetballGame events: ${error.message}`);

  // Mark complete so the stats aggregator includes it.
  await admin
    .from("games")
    .update({ status: "completed" })
    .eq("id", game.id);

  return { team, players, game, ownerId };
}

test("NETBALL-05: stats dashboard renders all 5 sections after a finalised netball game", async ({ page }) => {
  const { team } = await seedFinalisedNetballGame({ trackScoring: true });
  await page.goto(`/teams/${team.id}/stats`);

  // Hit each of the 5 section headings; if any fails the dashboard is

  const sectionTitles = [

    /attendance/i,
  ];
  for (const re of sectionTitles) {

  }

  // Per-position breakdown beneath the Attack column. With one game
  // played and the GS at lineup index 0, that player should have
  // "GS 100%" or similar under the Attack third cell.
  // Use a permissive regex — the exact percentage depends on whether
  // any other "go"-position breakdown logic kicks in.
  await expect(page.getByText(/GS \d+%/)).toBeVisible({ timeout: 5_000 });
});

test("NETBALL-05: stats dashboard does NOT render AFL aggregator headings", async ({ page }) => {
  const { team } = await seedFinalisedNetballGame({ trackScoring: true });
  await page.goto(`/teams/${team.id}/stats`);

  // The dashboard MUST be the netball shell — confirmed by netball-only
  // heading present.
  await expect(page.getByRole("heading", { name: /player chemistry/i })).toBeVisible();

  // AFL aggregator section headings — none of these should render on

  // AFL; NetballDashboardShell.tsx does not.)
  await expect(page.getByRole("heading", { name: /winning combinations/i })).not.toBeVisible();
  await expect(page.getByRole("heading", { name: /position fit/i })).not.toBeVisible();
  await expect(page.getByRole("heading", { name: /quarter scoring/i })).not.toBeVisible();
});
