STEP A — create the shared helper src/lib/live/reconcileLineupToAvailability.ts:

```typescript
import type { SupabaseClient } from "@supabase/supabase-js";

// Build the availableIds union (game_availability 'available' rows +
// game_fill_ins ids + player_arrived events) for a game, mirroring the
// live-read union in live/page.tsx:211-221 so "available at kickoff" ==
// "available live". Takes an ALREADY-RESOLVED client (resolveWriter is not
// exported; the three start actions each hold their `w.supabase`).
export async function buildAvailableIds(
  supabase: SupabaseClient,
  gameId: string,
): Promise<Set<string>> {
  const [{ data: avail }, { data: fillIns }, { data: events }] = await Promise.all([
    supabase.from("game_availability").select("player_id").eq("game_id", gameId).eq("status", "available"),
    supabase.from("game_fill_ins").select("id").eq("game_id", gameId),
    supabase.from("game_events").select("player_id, type").eq("game_id", gameId).eq("type", "player_arrived"),
  ]);
  return new Set<string>([
    ...((avail ?? []).map((a) => a.player_id as string)),
    ...((fillIns ?? []).map((f) => f.id as string)),
    ...((events ?? []).filter((e) => e.player_id).map((e) => e.player_id as string)),
  ]);
}

// Generic structural filter: any lineup is an object whose values are string[]
// arrays of player ids (AFL Lineup, LeagueLineup, netball GenericLineup all
// satisfy this). Strip every id not in availableIds; leave vacated spots empty
// (D-04: the normal rotation fills them — no backfill here).
export async function reconcileLineupToAvailability<T extends Record<string, string[]>>(
  supabase: SupabaseClient,
  gameId: string,
  lineup: T,
): Promise<T> {
  const availableIds = await buildAvailableIds(supabase, gameId);
  const out = {} as T;
  for (const key of Object.keys(lineup) as (keyof T)[]) {
    out[key] = (lineup[key] as string[]).filter((id) => availableIds.has(id)) as T[keyof T];
  }
  return out;
}
```
