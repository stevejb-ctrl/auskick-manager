// 1. Add imports (if not already present in the file):
import { getSportConfig } from "@/lib/sports";
import type { Sport } from "@/lib/types";

// 2. Replace each occurrence of:
//      `team.age_group as AgeGroup`  OR
//      `(team?.age_group ?? "U10") as AgeGroup`  OR
//      `AGE_GROUPS[team.age_group]`
// with:
const sport = ((team as { sport?: string | null } | null)?.sport ?? "afl") as Sport;
const ageGroupCfg = getSportConfig(sport).ageGroups.find(g => g.id === team.age_group)
  ?? getSportConfig(sport).ageGroups[0]; // fallback to first age group of that sport

// 3. Use ageGroupCfg.maxSquadSize / ageGroupCfg.periodSeconds / ageGroupCfg.positionModel
//    in place of any AGE_GROUPS[ageGroup].field reference.
